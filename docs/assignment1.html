Animated GIF creation

#Frames

drake_yes <- image_read("https://attachments.f95zone.to/2021/05/1202828_72494888.jpg")

drake_no <- image_read("https://static.hiphopdx.com/2015/10/drake-hotline-bling-jacket-moncler.png")

blank_yes <- image_blank(300, 300, color = "white") %>%
  image_annotate(text = "Make a meme\nfor your\nassignment:", size = 40, gravity = "center", color = "black")
  
blank_no <- image_blank(300, 300, color = "white") %>%
  image_annotate(text = "Do your\nassignment:", size = 40, gravity = "center", color = "black")

blank <- image_blank(300, 300, color = "white") #break off point for each cycle

#Animation

gif <- c(blank_no, drake_no, blank_yes, drake_yes, blank) %>%
  image_resize("300x300") %>%
    image_animate(fps = 0.5, dispose = "previous")

#Output

print(gif)
